Laravel框架下的php辅助函数

By yison

发表于 2019-03-22

Laravel框架自带了一系列的PHP辅助函数，很多被框架自身使用，多了解一点这些函数，也方便我们在代码中善用它们。

方法列表

数组函数

array_add()
功能：array_add 函数添加给定键值对到数组（如果给定键不存在的话）。
用法：

$array = array_add(['name' => 'Desk'], 'price', 100);
// ['name' => 'Desk', 'price' => 100]

array_collapse()
功能：array_collapse 函数将多个数组合并成一个。
用法：

$array = array_collapse([[1, 2, 3], [4, 5, 6], [7, 8, 9]]);
// [1, 2, 3, 4, 5, 6, 7, 8, 9]

array_divide()
功能：array_divide 函数返回两个数组，一个包含原数组的所有键，另外一个包含原数组的所有值。
用法：

list($keys, $values) = array_divide(['name' => 'Desk']);
// $keys: ['name']
// $values: ['Desk']

array_dot()
功能：array_dot 函数使用”.”号将将多维数组转化为一维数组。
用法：

$array = ['products' => ['desk' => ['price' => 100]]];
$flattened = array_dot($array);
// ['products.desk.price' => 100]

array_except()
功能：array_except 函数从数组中移除给定键值对。
用法：

$array = ['name' => 'Desk', 'price' => 100];
$array = array_except($array, ['price']);
// ['name' => 'Desk']

array_first()
功能：array_first 函数返回通过测试数组的第一个元素。
用法：

$array = [100, 200, 300];

$value = array_first($array, function ($value, $key) {
    return $value >= 150;
});
// 200

// 默认值可以作为第三个参数传递给该方法，如果没有值通过测试的话返回默认值：
$value = array_first($array, $callback, $default);

array_flatten()
功能：array_flatten 函数将多维数组转化为一维数组。
用法：

$array = ['name' => 'Joe', 'languages' => ['PHP', 'Ruby']];

$array = array_flatten($array);
// ['Joe', 'PHP', 'Ruby'];

array_forget()
功能：array_forget 函数使用”.”号从嵌套数组中移除给定键值对。
用法：

$array = ['products' => ['desk' => ['price' => 100]]];

array_forget($array, 'products.desk');
// ['products' => []]

array_get()
功能：array_get 方法使用”.”号从嵌套数组中获取值。
用法：

$array = ['products' => ['desk' => ['price' => 100]]];

$value = array_get($array, 'products.desk.price');
// ['price' => 100]

// array_get 函数还接收一个默认值，如果指定键不存在的话则返回该默认值：
$value = array_get($array, 'products.desk.discount', 0);
// 0

array_has()
功能：array_has 函数使用“.”检查给定数据项是否在数组中存在。
用法：

$array = ['product' => ['name' => 'desk', 'price' => 100]];

$hasItem = array_has($array, 'product.name');
// true

$hasItems = array_has($array, ['product.price', 'product.discount']);
// false

array_last()
功能：array_last 函数返回通过过滤数组的最后一个元素。
用法：

$array = [100, 200, 300, 110];

$value = array_last($array, function ($value, $key) {
    return $value >= 150;
});
// 300

// 我们可以传递一个默认值作为第三个参数到该函数，如果没有值通过真理测试的话该默认值被返回：
$last = array_last($array, $callback, $default);

array_only()
功能：array_only 方法只从给定数组中返回指定键值对。
用法：

$array = ['name' => 'Desk', 'price' => 100, 'orders' => 10];

$array = array_only($array, ['name', 'price']);
// ['name' => 'Desk', 'price' => 100]

array_pluck()
功能：array_pluck 方法从数组中返回给定键对应的键值对列表。
用法：

$array = [
    ['developer' => ['id' => 1, 'name' => 'Taylor']],
    ['developer' => ['id' => 2, 'name' => 'Abigail']],
];

$names = array_pluck($array, 'developer.name');
// ['Taylor', 'Abigail']

// 你还可以指定返回结果的键：
$array = array_pluck($array, 'developer.name', 'developer.id');
// [1 => 'Taylor', 2 => 'Abigail'];

array_prepend()
功能：array_prepend 函数将数据项推入数组开头。
用法：

$array = ['one', 'two', 'three', 'four'];

$array = array_prepend($array, 'zero');
// $array: ['zero', 'one', 'two', 'three', 'four']

// 如果需要的话还可以指定用于该值的键：
$array = ['price' => 100];

$array = array_prepend($array, 'Desk', 'name');
// ['name' => 'Desk', 'price' => 100]

array_pull()
功能：array_pull 函数从数组中返回并移除键值对。
用法：

$array = ['name' => 'Desk', 'price' => 100];

$name = array_pull($array, 'name');
// $name: Desk
// $array: ['price' => 100]

// 我们还可以传递默认值作为第三个参数到该函数，如果指定键不存在的话返回该值：
$value = array_pull($array, $key, $default);

array_random()
功能：array_random 函数从数组中返回随机值。
用法：

$array = [1, 2, 3, 4, 5];

$random = array_random($array);
// 4 - (retrieved randomly)

// 还可以指定返回的数据项数目作为可选的第二个参数，需要注意的是提供这个参数会返回一个数组，即使只返回一个数据项：
$items = array_random($array, 2);
// [2, 5] - (retrieved randomly)

array_set()
功能：array_set 函数用于在嵌套数组中使用”.”号设置值。
用法：

$array = ['products' => ['desk' => ['price' => 100]]];

array_set($array, 'products.desk.price', 200);
// ['products' => ['desk' => ['price' => 200]]]

array_sort()
功能：array_sort 函数通过值对数组进行排序。
用法：

$array = ['Desk', 'Table', 'Chair'];

$sorted = array_sort($array);
// ['Chair', 'Desk', 'Table']

// 还可以通过给定闭包的结果对数组进行排序：
$array = [
    ['name' => 'Desk'],
    ['name' => 'Table'],
    ['name' => 'Chair'],
];

$sorted = array_values(array_sort($array, function ($value) {
    return $value['name'];
}));
/*
    [
        ['name' => 'Chair'],
        ['name' => 'Desk'],
        ['name' => 'Table'],
    ]
*/

array_sort_recursive()
功能：array_sort_recursive 函数使用 sort 函数对数组进行递归排序。
用法：

$array = [
    ['Roman', 'Taylor', 'Li'],
    ['PHP', 'Ruby', 'JavaScript'],
];

$array = array_sort_recursive($array);
/*
    [
        ['Li', 'Roman', 'Taylor'],
        ['JavaScript', 'PHP', 'Ruby'],
    ]
*/

array_where()
功能：array_where 函数使用给定闭包对数组进行过滤。
用法：

$array = [100, '200', 300, '400', 500];

$array = array_where($array, function ($value, $key) {
    return is_string($value);
});
// [1 => 200, 3 => 400]

array_wrap()
功能：array_wrap 函数将给定值包裹到数组中，如果给定值已经是数组则保持不变。
用法：

$string = 'Laravel';

$array = array_wrap($string);
// ['Laravel']

// 如果给定值是空的，则返回一个空数组：
$nothing = null;

$array = array_wrap($nothing);
// []

data_fill()
功能：data_fill 函数使用「.」号以嵌套数组或对象的方式设置缺失值。
用法：

$data = ['products' => ['desk' => ['price' => 100]]];

data_fill($data, 'products.desk.price', 200);
// ['products' => ['desk' => ['price' => 100]]]

data_fill($data, 'products.desk.discount', 10);
// ['products' => ['desk' => ['price' => 100, 'discount' => 10]]]

// 该函数还接收「*」号作为通配符并填充相应目标：
$data = [
    'products' => [
        ['name' => 'Desk 1', 'price' => 100],
        ['name' => 'Desk 2'],
    ],
];

data_fill($data, 'products.*.price', 200);
/*
    [
        'products' => [
            ['name' => 'Desk 1', 'price' => 100],
            ['name' => 'Desk 2', 'price' => 200],
        ],
    ]
*/

data_get()
功能：data_get 函数使用「.」号从嵌套数组或对象中获取值。
用法：

$data = ['products' => ['desk' => ['price' => 100]]];

$price = data_get($data, 'products.desk.price');
// 100

data_get 函数还接收默认值，以便指定键不存在的情况下返回：
$discount = data_get($data, 'products.desk.discount', 0);
// 0

data_set()
功能：data_set 函数使用 「.」号设置嵌套数组或对象的值。
用法：

$data = ['products' => ['desk' => ['price' => 100]]];

data_set($data, 'products.desk.price', 200);
// ['products' => ['desk' => ['price' => 200]]]

// 该函数还接收通配符然后设置相应的目标值：
$data = [
    'products' => [
        ['name' => 'Desk 1', 'price' => 100],
        ['name' => 'Desk 2', 'price' => 150],
    ],
];

data_set($data, 'products.*.price', 200);
/*
    [
        'products' => [
            ['name' => 'Desk 1', 'price' => 200],
            ['name' => 'Desk 2', 'price' => 200],
        ],
    ]
*/

// 默认情况下，任意已存在的值都会被覆盖，如果你想要只设置不存在的值，可以传递 false 作为第三个参数：
$data = ['products' => ['desk' => ['price' => 100]]];

data_set($data, 'products.desk.price', 200, false);
// ['products' => ['desk' => ['price' => 100]]]

head()
head 函数只是简单返回给定数组的第一个元素。
用法：

$array = [100, 200, 300];

$first = head($array);
// 100

last()
功能：last 函数返回给定数组的最后一个元素。
用法：

$array = [100, 200, 300];

$last = last($array);
// 300

路径函数

app_path()
功能：app_path 函数返回 app 目录的绝对路径，你还可以使用 app_path 函数为相对于 app 目录的给定文件生成绝对路径。
用法：

$path = app_path();
$path = app_path('Http/Controllers/Controller.php');

base_path()
功能：base_path 函数返回项目根目录的绝对路径，你还可以使用 base_path 函数为相对于应用根目录的给定文件生成绝对路径。
用法：

$path = base_path();
$path = base_path('vendor/bin');

config_path()
功能：config_path 函数返回应用配置目录 config 的绝对路径，还可以使用 config_path 函数在应用配置目录内为给定文件生成完整路径。
用法：

$path = config_path();

$path = config_path('app.php');

database_path()
功能：database_path 函数返回应用数据库目录 database 的完整路径，还可以使用 database_path 函数在数据库目录内为给定文件生成完整路径。
用法：

$path = database_path();

$path = database_path('factories/UserFactory.php');

mix() 功能：mix 函数返回带有版本号的Mix文件路径。
用法：

mix($file);

public_path()
功能：public_path 函数返回 public 目录的绝对路径，还可以使用 public_path 函数在 public 目录下为给定文件生成完整路径。
用法：

$path = public_path();

$path = public_path('css/app.css');

resource_path()
功能：resource_path 函数返回 resources 目录的绝对路径，还可以使用 resources 函数在 resources 目录下为给定文件生成完整路径。
用法：

$path = resource_path();

$path = resource_path('assets/sass/app.scss');

storage_path() 功能：storage_path 函数返回 storage 目录的绝对路径， 还可以使用 storage_path 函数在 storage 目录下为给定文件生成完整路径。
用法：

$path = storage_path();

$path = storage_path('app/file.txt');

字符串函数

__()
功能：__ 函数会使用本地化文件翻译给定翻译字符串或翻译键。
用法：

echo __('Welcome to our application');

echo __('messages.welcome');

// 如果给定翻译字符串或键不存在，__ 函数将会返回给定值。  
// 所以，使用上面的例子，如果翻译键不存在的话 __ 函数将会返回 messages.welcome。

camel_case()
功能：camel_case 函数将给定字符串转化为符合驼峰式命名规则的字符串。
用法：

$camel = camel_case('foo_bar');
// fooBar

class_basename()
功能：class_basename 返回给定类移除命名空间后的类名。
用法：

$class = class_basename('Foo\Bar\Baz');
// Baz

e()
功能：e 函数在给定字符串上运行 htmlentities（double_encode 选项设置为 false）。
用法：

echo e('<html>foo</html>');
// &lt;html&gt;foo&lt;/html&gt;

ends_with()
功能：ends_with 函数判断给定字符串是否以给定值结尾。
用法：

$value = ends_with('This is my name', 'name');
// true

kebab_case()
功能：kebab_case 函数将给定字符串转化为短划线分隔的字符串。
用法：

$converted = kebab_case('fooBar');
// foo-bar

preg_replace_array()
功能：preg_replace_array 函数使用数组替换字符串序列中的给定模式。
用法：

$string = 'The event will take place between :start and :end';

$replaced = preg_replace_array('/:[a-z_]+/', ['8:30', '9:00'], $string);
// The event will take place between 8:30 and 9:00

snake_case()
功能：snake_case 函数将给定字符串转化为下划线分隔的字符串。
用法：

$snake = snake_case('fooBar');
// foo_bar

starts_with()
功能：starts_with 函数判断给定字符串是否以给定值开头。
用法：

$result = starts_with('This is my name', 'This');
// true

str_after()
功能：str_after 函数返回字符串中给定值之后的所有字符。
用法：

$slice = str_after('This is my name', 'This is');
// ' my name'

str_before()
功能：str_before 函数返回字符串给定值之前的所有字符。
用法：

$slice = str_before('This is my name', 'my name');
// 'This is '

str_contains()
功能：str_contains 函数判断给定字符串是否包含给定值（大小写敏感）。
用法：

$contains = str_contains('This is my name', 'my');
// true

// 还可以传递数组值判断给定字符串是否包含数组中的任意值：
$contains = str_contains('This is my name', ['my', 'foo']);
// true

str_finish()
功能：str_finish 函数添加给定值单个实例到字符串结尾 —— 如果原字符串不以给定值结尾的话。
用法：

$adjusted = str_finish('this/string', '/');
// this/string/

$adjusted = str_finish('this/string/', '/');
// this/string/

str_is()
功能：str_is 函数判断给定字符串是否与给定模式匹配，星号可用于表示通配符。
用法：

$value = str_is('foo*', 'foobar');
// true

$value = str_is('baz*', 'foobar');
// false

str_limit()
功能：str_limit 函数以指定长度截断字符串。
用法：

$truncated = str_limit('The quick brown fox jumps over the lazy dog', 20);
// The quick brown fox...

// 还可以传递第三个参数来改变字符串末尾字符：
$truncated = str_limit('The quick brown fox jumps over the lazy dog', 20, ' (...)');
// The quick brown fox (...)

str_plural()
功能：str_plural 函数将字符串转化为复数形式，该函数当前只支持英文。
用法：

$plural = str_plural('car');
// cars

$plural = str_plural('child');
// children

// 还可以传递整型数据作为第二个参数到该函数以获取字符串的单数或复数形式：
$plural = str_plural('child', 2);
// children

$plural = str_plural('child', 1);
// child

str_random()
功能：str_random 函数通过指定长度生成随机字符串，该函数使用了PHP的 random_bytes 函数。
用法：

$string = str_random(40);

str_replace_array()
功能：str_replace_array 函数使用数组在字符串序列中替换给定值。
用法：

$string = 'The event will take place between ? and ?';

$replaced = str_replace_array('?', ['8:30', '9:00'], $string);
// The event will take place between 8:30 and 9:00

str_replace_first()
功能：str_replace_first 函数会替换字符串中第一次出现的值。
用法：

$replaced = str_replace_first('the', 'a', 'the quick brown fox jumps over the lazy dog');
// a quick brown fox jumps over the lazy dog

str_replace_last()
功能：str_replace_last 函数会替换字符串中最后一次出现的值。
用法：

$replaced = str_replace_last('the', 'a', 'the quick brown fox jumps over the lazy dog');
// the quick brown fox jumps over a lazy dog

str_singular()
功能：str_singular 函数将字符串转化为单数形式，该函数目前只支持英文。
用法：

$singular = str_singular('cars');
// car

$singular = str_singular('children');
// child

str_slug()
功能：str_slug 函数将给定字符串生成 URL 友好的格式。
用法：

$title = str_slug("Laravel 5 Framework", "-");
// laravel-5-framework

str_start()
功能：如果字符串没有以给定值开头的话 str_start 函数会将给定值添加到字符串最前面。
用法：

$adjusted = str_start('this/string', '/');
// /this/string

$adjusted = str_start('/this/string/', '/');
// /this/string

studly_case()
功能：studly_case 函数将给定字符串转化为单词开头字母大写的格式。
用法：

$value = studly_case('foo_bar');
// FooBar

title_case()
功能：title_case 函数将字符串转化为 Title 形式。
用法：

$title = title_case('a nice title uses the correct case');
// A Nice Title Uses The Correct Case

trans()
功能：trans 函数使用本地文件翻译给定翻译键。
用法：

echo trans('messages.welcome');
// 如果指定翻译键不存在，trans 函数会返回给定键。
// 所以，以上面的示例为例，如果翻译键不存在，trans 函数会返回 messages.welcome。

trans_choice()
功能：trans_choice 函数翻译带拐点的给定翻译键。
用法：

echo trans_choice('messages.notifications', $unreadCount);
// 如果指定的翻译键不存在，trans_choice 函数会将其返回。
// 所以，以上面的示例为例，如果指定翻译键不存在 trans_choice 函数会返回 messages.notifications。

URL函数

action()
功能：action 函数为给定控制器动作生成 URL，你不需要传递完整的命名空间到该控制器，传递相对于命名空间 App\Http\Controllers 的类名即可。
用法：

$url = action('HomeController@index');

// 如果该方法接收路由参数，你可以将其作为第二个参数传递进来：
$url = action('UserController@profile', ['id' => 1]);

asset()
功能：asset 函数使用当前请求的 scheme（HTTP 或 HTTPS）为前端资源生成一个 URL。
用法：

$url = asset('img/photo.jpg');

secure_asset()
功能：secure_asset 函数使用 HTTPS 为前端资源生成一个 URL。
用法：

echo secure_asset('foo/bar.zip', $title, $attributes = []);

route()
功能：route 函数为给定命名路由生成一个URL。
用法：

$url = route('routeName');

// 如果该路由接收参数，你可以将其作为第二个参数传递进来：
$url = route('routeName', ['id' => 1]);

// 默认情况下，route 函数生成的是绝对 URL，如果你想要生成一个相对 URL，可以传递 false 作为第三个参数：
$url = route('routeName', ['id' => 1], false);

secure_url()
功能：secure_url 函数为给定路径生成完整的 HTTPS URL。
用法：

echo secure_url('user/profile');

echo secure_url('user/profile', [1]);

url()
功能：url 函数为给定路径生成完整URL。
用法：

echo url('user/profile');

echo url('user/profile', [1]);

// 如果没有提供路径，将会返回 Illuminate\Routing\UrlGenerator 实例：
echo url()->current();
echo url()->full();
echo url()->previous();

其它函数

abort()
功能：abort 函数会抛出一个被异常处理器渲染的 HTTP 异常。
用法：

abort(403);

// 还可以提供异常响应文本以及自定义响应头：
abort(403, 'Unauthorized.', $headers);

abort_if()
功能：abort_if 函数在给定布尔表达式为 true 时抛出 HTTP 异常。
用法：

abort_if(! Auth::user()->isAdmin(), 403);
// 和 abort 一样，你还可以传递异常响应文本作为第三个参数以及自定义响应头数组作为第四个参数。

abort_unless()
功能：abort_unless 函数在给定布尔表达式为 false 时抛出 HTTP 异常。
用法：

abort_unless(Auth::user()->isAdmin(), 403);

// 和 abort 一样，你还可以传递异常响应文本作为第三个参数以及自定义响应头数组作为第四个参数。

app()
功能：app 函数返回服务容器实例。
用法：

$container = app();

功能：rescue 函数可以执行给定闭包并捕获执行过程中的所有异常。这些捕获的异常会发送给异常处理器的 report 方法，不过，请求会继续执行。
用法：

return rescue(function () {
    return $this->method();
});

// 还可以传递第二个参数到 rescue 函数，作为在执行闭包出现异常的情况下返回的默认值：

});

retry()
功能：retry 函数尝试执行给定回调直到达到最大执行次数，如果回调没有抛出异常，会返回对应的返回值。如果回调抛出了异常，会自动重试。如果超出最大执行次数，异常会被抛出。
用法：

return retry(5, function () {
    // Attempt 5 times while resting 100ms in between attempts...
}, 100);

session()
功能：session 函数可以用于获取/设置 Session 值。

tap()
功能：tap 函数接收两个参数：任意的 $value 和一个闭包。$value 会被传递到闭包然后通过 tap 函数返回。闭包返回值与函数返回值不相关。
用法：

$user = tap(User::first(), function ($user) {
    $user->name = 'taylor';

    $user->save();
});

// 如果没有传入闭包到 tap 函数，那么你可以调用给定 $value 上面的任意方法，调用方法的返回值永远都是 $value，不管在方法中定义的返回值是什么。
// 例如，Eloquent update 方法通常返回一个整型，不过，我们可以通过 tap 函数强制该方法返回模型本身：
$user = tap($user)->update([
    'name' => $name,
    'email' => $email,
]);

);

with()
功能：with 函数返回给定的值，如果第二个参数是闭包，则返回闭包执行结果。
用法：

$callback = function ($value) {
    return (is_numeric($value)) ? $value * 2 : 0;
};
$result = with(5, $callback);
// 10

$result = with(null, $callback);
// 0

$result = with(5, null);
// 5
